# 1. Buat script anti crypto miner (copy semua baris ini sekaligus)
cat > ~/kill-miner.sh << 'EOF'
#!/bin/bash
ps aux | grep "/dev/shm" | grep -v grep | awk '{print $2}' | xargs -r kill -9 2>/dev/null
rm -f /dev/shm/8Bxsuye /dev/shm/JJdNwbR /dev/shm/mxDbtFcvk /dev/shm/S8
find /dev/shm -type f -executable -mmin +5 -delete 2>/dev/null
EOF

chmod +x ~/kill-miner.sh

# 2. Setup auto-run setiap 5 menit
(crontab -l 2>/dev/null; echo "*/5 * * * * /home/deploy/kill-miner.sh") | crontab -

# 3. Cek apakah berhasil
crontab -l

# 4. Test jalankan manual
~/kill-miner.sh
