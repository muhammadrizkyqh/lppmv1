# === SETUP NGINX + DOMAIN stai-ali.com ===
# Copy-paste di VPS (sudah login ssh deploy@195.88.211.123)

# 1. Install Nginx
sudo apt update
sudo apt install -y nginx

# 2. Start Nginx
sudo systemctl start nginx
sudo systemctl enable nginx

# 3. Buat config Nginx untuk stai-ali.com
sudo tee /etc/nginx/sites-available/stai-ali << 'EOF'
server {
    listen 80;
    server_name stai-ali.com www.stai-ali.com;

    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_buffering off;
        proxy_read_timeout 300s;
        proxy_connect_timeout 75s;
    }

    location /_next/static {
        proxy_pass http://127.0.0.1:3000;
        proxy_cache_valid 60m;
        add_header Cache-Control "public, immutable";
    }
}
EOF

# 4. Enable config
sudo ln -sf /etc/nginx/sites-available/stai-ali /etc/nginx/sites-enabled/
sudo rm -f /etc/nginx/sites-enabled/default

# 5. Test config
sudo nginx -t

# 6. Restart Nginx
sudo systemctl restart nginx

# 7. Install Certbot untuk SSL
sudo apt install -y certbot python3-certbot-nginx

# 8. Generate SSL certificate untuk stai-ali.com
sudo certbot --nginx -d stai-ali.com -d www.stai-ali.com

# Certbot akan tanya:
# - Email: [masukkan email Anda]
# - Terms: Yes (Y)
# - Share email: No (N)
# - Redirect HTTP to HTTPS: Yes (2)

# 9. Verify SSL
sudo certbot certificates

# 10. Test auto-renewal
sudo certbot renew --dry-run

# SELESAI!
# Website accessible di:
# - https://stai-ali.com (SSL active)
# - https://www.stai-ali.com
# Auto-redirect dari HTTP ke HTTPS
